from django.db import models

class Company(models.Model):
    """Represents a company entity."""
    name = models.CharField(max_length=255, verbose_name='Név')
    description = models.TextField(verbose_name='Leírás')

    def __str__(self):
        return self.name

class Team(models.Model):
    """Represents a team within a company."""
    company = models.ForeignKey(Company, on_delete=models.CASCADE, verbose_name='Vállalat')
    name = models.CharField(max_length=255, verbose_name='Név')
    description = models.TextField(verbose_name='Leírás')

    def __str__(self):
        return self.name

class Squad(models.Model):
    """Represents a squad within a team."""
    team = models.ForeignKey(Team, on_delete=models.CASCADE, verbose_name='Csapat')
    name = models.CharField(max_length=255, verbose_name='Név')
    description = models.TextField(verbose_name='Leírás')

    def __str__(self):
        return self.name

class Position(models.Model):
    """Represents a position within the organization."""
    name = models.CharField(max_length=255, verbose_name='Név')
    description = models.TextField(verbose_name='Leírás')

    def __str__(self):
        return self.name

class Employee(models.Model):
    """Represents an employee with skills and a position within a squad."""
    squad = models.ForeignKey(Squad, on_delete=models.CASCADE, verbose_name='Squad')
    name = models.CharField(max_length=255, verbose_name='Név')
    position = models.ForeignKey(Position, on_delete=models.CASCADE, verbose_name='Pozíció')

    def __str__(self):
        return self.name

class Skill(models.Model):
    """Represents a skill of an employee."""
    name = models.CharField(max_length=255, verbose_name='Név')
    level = models.PositiveIntegerField(default=1, verbose_name='Szint')
    employee = models.ForeignKey(Employee, related_name='skills', on_delete=models.CASCADE, verbose_name='Kolléga')

    def __str__(self):
        return f"{self.name} (Szint: {self.level})"
